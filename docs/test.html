Use the console.

Some examples: Note that `type` most happen before `autocomplete` and `documentation`.

shared.postMessage({msgType: "type", code: "module List2 = struct let it a = a + 2 end", msgId: 1});
shared.postMessage({msgType: "execute", code: "module List2 = struct let it a = a + 2 end", msgId: 1});
shared.postMessage({msgType: "autocomplete", msgId: 1, posFname: "xxx", posLnum: 0, posBol: 0, posCnum: 0, text: "List.it", showDocs:true});
shared.postMessage({msgType: "documentation", msgId: 1, posFname: "xxx", posLnum: 0, posBol: 0, posCnum: 0, text: "List.iteri"})
